MODELS

staging/stg_prices.sql

materialized: table 
SELECT
  CAST(product_id AS STRING),
  CAST(name AS STRING),
  CAST(price AS FLOAT64),
  CAST(source AS STRING),
  CAST(scraped_at AS TIMESTAMP)
FROM {{ source('bigtable','prices') }}
Columns: product_id, name, price,
old_price, discount_pct, rating,
brand, category, source, scraped_at

cleaned/prices_cleaned.sql

materialized: table 
SELECT
  product_id,
  TRIM(name) as name,
  COALESCE(brand,'Unknown'),
  price,
  ROUND(discount_pct, 2),
  rating, category, source,
  scraped_at
FROM {{ ref('stg_prices') }}
WHERE price > 0
  AND name IS NOT NULL
Tests: not_null, unique (product_id)
accepted_values: source
5,739 records after cleaning

aggregated/prices_aggregated.sql

materialized: table 
SELECT
  source, category,
  COUNT(*) as total_products,
  ROUND(AVG(price),2) as avg_price,
  MIN(price) as min_price,
  MAX(price) as max_price,
  ROUND(AVG(discount_pct),2)
FROM {{ ref('prices_cleaned') }}
GROUP BY source, category
Output for: Streamlit dashboard
+ Python stats notebook
